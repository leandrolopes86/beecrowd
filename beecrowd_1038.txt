using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Globalization;
using System.Runtime.Serialization;
//using System.Net.Http.Headers;

public class beecrowd
{
    public static void Main(string[] args)
    {
        int qtd = 0; double valor1 = 0, valor2 = 0, valor3 = 0, valor4 = 0, valor5 = 0, valorTotal; double cachorro_quente = 4.00, x_salada = 4.50, x_bacon = 5.00, torrada_simples = 2.00, refrigerante = 1.50;

        string[] e = Console.ReadLine().Split(' ');

        char codigo = Convert.ToChar(e[0]);
        qtd = Convert.ToInt32(e[1]);

        switch (codigo)
        {
            case '1':
                valor1 = cachorro_quente * qtd;
                break;
            case '2':
                valor2 = x_salada * qtd;
                break;
            case '3':
                valor3 = x_bacon * qtd;
                break;
            case '4':
                valor4 = torrada_simples * qtd;
                break;
            case '5':
                valor5 = refrigerante * qtd;
                break;
        }

        valorTotal = valor1 + valor2 + valor3 + valor4 + valor5;

        Console.WriteLine("Total: R$ " + valorTotal.ToString("F2"));


    }
}
